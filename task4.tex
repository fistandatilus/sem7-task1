\begin{equation} \label{p3}
    \begin{array}{l}
        \rho_0(x) = 1, u_0 = 0, x \in [0;10],\\
        u(t,0) = \tilde{v}, \rho(t, 0) = \tilde{\rho}, \pd{u}{x}\vert_{x=10} = 0, t \in [0; T]. 
    \end{array}
\end{equation}
